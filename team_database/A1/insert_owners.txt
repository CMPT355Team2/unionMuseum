from ( values (
	:nswscode,

	:nswswname, 
	:nswsregion, 
	:nswscentury, 
	:nswscategory,

	:nswscdatecom, 
	:nswsadateacq,

	:nswsstatus,
	:nswsinsvalue,
	:nswsinsname
)) AS t 
(newcode, newacronym, newmname, newwname, newregion, newcentury, newcategory, newauthor, newcdatecom, newadateacq, newwdesc, newstatus, newinsvalue, newinsname);


insert into 
	works

	tw.newauthor,
	tw.newcdatecom,
